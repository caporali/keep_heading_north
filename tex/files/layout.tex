% amsthm
	\newtheorem{thm}{Theorem}
	\newtheorem*{thm*}{Theorem}
	\crefname{thm}{Theorem}{Theorems}
	\theoremstyle{remark}
	\newtheorem{rem}{Remark}
	\newtheorem*{rem*}{Remark}
	\crefname{rem}{Remark}{Remarks}

% fonts
	\titleformat{\section}{\Large\bfseries\scshape}{\thesection.}{0.5em}{}
	\titleformat{\subsection}{\large\bfseries\scshape}{\thesubsection.}{0.5em}{}
	\titleformat{\subsubsection}{\normalsize\bfseries\scshape}{\thesubsubsection.}{0.5em}{}

% title
	\newcommand{\definetitle}[2]{
		\title{{\scshape #1}\vspace{-0.5cm}}
		\author{{\scshape #2}}
		\date{\vspace{-0.35cm}{\scshape \today}}
		\setlength{\droptitle}{-5em}
	}

% abstract
	\newcommand{\defineabstract}[1]{
		\begin{center}
			\vspace{-0.75cm}
			\line(1, 0){375}
				\renewcommand{\abstractname}{\bfseries\scshape Abstract \vspace{0.15cm}}
				\begin{abstract}
					\noindent #1
				\end{abstract}
				\vspace{-0.25cm}
			\line(1, 0){375}
		\end{center}
	}
